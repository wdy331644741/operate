CREATE TABLE `promoter_list` (
  `apply_id` int(10) unsigned NOT NULL AUTO_INCREMENT COMMENT '申请id',
  `auth_id` int(10) NOT NULL COMMENT '用户id',
  `username` varchar(30) NOT NULL COMMENT '用户姓名',
  `phone` char(11) NOT NULL COMMENT '电话',
  `invite_num` int(5) NOT NULL COMMENT '邀请用户数量',
  `total_inve_amount` decimal(20,2) NOT NULL COMMENT '好友投资总额',
  `commission` decimal(20,2) NOT NULL COMMENT '好友邀请所得佣金',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',
  `create_time` datetime NOT NULL COMMENT '创建时间',
  `status` tinyint(1) NOT NULL DEFAULT '0' COMMENT '0待审核，1通过,2不通过',
  `level` tinyint(2) DEFAULT '1' COMMENT '等级',
  `earnings` decimal(20,2) DEFAULT '20.00' COMMENT '共享收益比例',
  PRIMARY KEY (`apply_id`)
) ENGINE=InnoDB AUTO_INCREMENT=140 DEFAULT CHARSET=utf8 COMMENT='推广员申请表';

CREATE TABLE `promoter_statistics` (
  `id` int(10) NOT NULL AUTO_INCREMENT,
  `date` date DEFAULT NULL COMMENT '日期',
  `adds` int(10) DEFAULT '0' COMMENT '推广员每日新增',
  `register_adds` int(10) DEFAULT '0' COMMENT '推广员渠道每日新增',
  `recharge_adds` int(10) DEFAULT '0' COMMENT '推广员渠道每日充值',
  `withdraw_adds` int(10) DEFAULT '0' COMMENT '推广员渠道每日提现',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='推广员相关数据统计';
